AWSTemplateFormatVersion: '2010-09-09'
Description: ECS Task Definition + single ALB Target Group for an app (prepared for rolling updates)

Parameters:
  ClusterName:
    Type: String
    Description: The name of the ECS Cluster stack from which to import Name.
  ServiceName:
    Type: String
  EnvFileS3Arn:
    Type: String
  VPCStackName:
    Type: String
    Description: The name of the vpc stack from which to import VPCID .
  Subnets:
    Type: List<AWS::EC2::Subnet::Id>
  SecurityGroupId:
    Type: AWS::EC2::SecurityGroup::Id
  ContainerName:
    Type: String
    Default: app
  ALBStackName:
    Type: String
    Description: "ARN of the existing ALB listener (port 80)"

Resources:

  ECSTaskExecutionRole:
    Type: AWS::IAM::Role
    Properties:
      RoleName: !Sub "${ServiceName}-ecsTaskExecutionRole"
      AssumeRolePolicyDocument:
        Version: '2012-10-17'
        Statement:
          - Effect: Allow
            Principal:
              Service: ecs-tasks.amazonaws.com
            Action: sts:AssumeRole
      Policies:
        - PolicyName: ECSExecutionPolicy
          PolicyDocument:
            Version: '2012-10-17'
            Statement:
              - Effect: Allow
                Action:
                  - ecr:GetAuthorizationToken
                  - ecr:BatchCheckLayerAvailability
                  - ecr:GetDownloadUrlForLayer
                  - ecr:BatchGetImage
                Resource: "*"
              - Effect: Allow
                Action:
                  - s3:GetObject
                Resource: !Ref EnvFileS3Arn
              - Effect: Allow
                Action:
                  - logs:CreateLogStream
                  - logs:PutLogEvents
                  - logs:CreateLogGroup
                Resource: "*"

  TaskDefinition:
    Type: AWS::ECS::TaskDefinition
    Properties:
      Family: !Sub "${ServiceName}"
      Cpu: '256'
      Memory: '512'
      NetworkMode: awsvpc
      RequiresCompatibilities:
        - FARGATE
      ExecutionRoleArn: !GetAtt ECSTaskExecutionRole.Arn
      ContainerDefinitions:
        - Name: !Ref ContainerName
          Image: nginx:latest
          PortMappings:
            - ContainerPort: 80
          EnvironmentFiles:
            - Value: !Ref EnvFileS3Arn
              Type: s3

  ALBTargetGroup:
    Type: AWS::ElasticLoadBalancingV2::TargetGroup
    Properties:
      Name: !Sub "${ServiceName}-tg"
      VpcId: !ImportValue
        Fn::Sub: "${VPCStackName}-VpcId"
      Protocol: HTTP
      Port: 80
      TargetType: ip
      HealthCheckPath: /

  ALBListenerRule:
    Type: AWS::ElasticLoadBalancingV2::ListenerRule
    Properties:
      ListenerArn: !ImportValue
        Fn::Sub: "${ALBStackName}-ListenerArn"
      Priority: 1
      Conditions:
        - Field: path-pattern
          Values:
            - "/*"
      Actions:
        - Type: forward
          TargetGroupArn: !Ref ALBTargetGroup

Outputs:
  TaskDefinitionArn:
    Description: "ECS Task Definition ARN"
    Value: !Ref TaskDefinition
    Export:
      Name: !Sub "${AWS::StackName}-TaskDefinitionArn"

  ALBTargetGroupArn:
    Description: "ALB Target Group ARN"
    Value: !Ref ALBTargetGroup
    Export:
      Name: !Sub "${AWS::StackName}-ALBTargetGroupArn"
